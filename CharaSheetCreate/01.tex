\newcommand*\circled[2][2pt]{
	\tikz
	[
		decoration={
			random steps,
			amplitude=1pt,
			segment length=5pt
		},
		baseline=(char.base)
	]{
		\node
		[
			decorate,
			shape=circle,
			draw,
			inner sep=#1-1pt
		]
		(char) {#2};
	}
}
